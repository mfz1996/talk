    private int test(int x, int y) {
        int width = Math.max(Math.abs(x), Math.abs(y)) * 2 + 1;
        int maxVal = width * width;
        int result;
        if (x >= 0 && y >= 0) {
            result = maxVal - 3 * (width - 1) - x + y;
        } else if (x <= 0 && y >= 0) {
            result = maxVal - 2 * (width - 1) - x - y;
        } else if (x <= 0 && y < 0) {
            result = maxVal - 1 * (width - 1) + x - y;
        } else {
            result = maxVal + x + y;
        }
        return result;
    }
